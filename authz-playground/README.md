# To set up 

$ cd authz-playground
$ python3 -m venv ./.venv && source .venv/bin/activate && pip install -r requirements.txt 

# To Run

In two different tabs

$ cd authz-playground
$ source .venv/bin/activate
$ rbt dev run

$ cd authz-playground
$ source .venv/bin/activate
$ python3 frontend/tester.py
